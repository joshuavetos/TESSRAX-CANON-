VK-01 does not negotiate.
VK-01 does not infer.

pub fn execute_vk01(evidence: VK01Inputs) -> TerminalState {
    if evidence.default_date.is_none() || evidence.last_action_date.is_none() {
        return TerminalState::HaltAmbiguity("MISSING_CHRONOLOGY");
    }

    if !evidence.has_receipt_chain {
        return TerminalState::HaltRejected("INV-U-D_BREACH");
    }

    let six_years_ms: i64 = 189_216_000_000;
    let delta = evidence.current_time - evidence.last_action_date.unwrap();

    if delta > six_years_ms {
        return TerminalState::HaltRejected("STATUTORY_STALENESS_DETECTED");
    }

    TerminalState::PassAuthority
}
